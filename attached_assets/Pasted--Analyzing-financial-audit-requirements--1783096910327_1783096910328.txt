القسم الأول: تعريف الجداول والـ Types المالية
1. DrawerTx — حركة الصندوق
type DrawerTx = {
  id: number;
  type: "in" | "out";      // داخل / خارج
  title: string;            // البيان
  category: string;         // التصنيف
  beneficiary?: string;     // المستفيد
  amount: number;           // المبلغ
  balance: number;          // الرصيد بعد العملية
  time: string;             // الوقت
  date: string;             // التاريخ "dd/mm/yyyy"
  auto?: boolean;
}

مصدر البيانات: drawers[deptId].txs[] في App state + جدول drawer_transactions في DB

2. DrawerState — صندوق القسم
type DrawerState = {
  balance: number;               // الرصيد الحالي
  txs: DrawerTx[];               // سجل الحركات
  openingBalance?: number;       // الرصيد الافتتاحي
  openingBalanceDate?: string;
}

مصدر البيانات: drawers: Record<string, DrawerState> — مفتاحه deptId
جداول DB: drawers (balance) + drawer_transactions (txs)

القسم الثاني: المعادلات المالية — واحدة بواحدة
① doDeposit — إيداع في الصندوق
الكود	newBal = currentBal + amount
المصدر	drawersRef.current[dept].balance
التأثير	يُضيف DrawerTx(type="in") + يُحدّث balance + يُرسل للـ API
مثال	رصيد=500، إيداع=100 → رصيد جديد=600 ✅
② doWithdraw — سحب من الصندوق
الكود	newBal = currentBal - amount (مع شرط: currentBal >= amount)
المصدر	drawersRef.current[dept].balance
التأثير	يُضيف DrawerTx(type="out") + يُحدّث balance
مثال	رصيد=500، سحب=100 → رصيد جديد=400 ✅
حماية	إذا amount > balance → لا شيء يحدث (صامت)
③ doAdjustBalance — تعديل رصيد يدوي
الكود	diff = newBalance - currentBalance → type = diff≥0 ? "in" : "out"
مثال	رصيد=500، تعديل إلى 300 → diff=-200، يُسجّل سحب 200 ✅
④ سعر الجلسة عند تسجيل مريض جديد (NewPatientScreen)

              ? basePrice × discount% / 100
              : parseFloat(form.discount) || 0

sessionPaid = parseFloat(form.paid) || 0
sessionDebt = Math.max(0, sessionNet - sessionPaid)

سيناريو	basePrice	خصم	paid	sessionDebt	صحيح؟
كامل	100	0	100	0	✅
جزئي	100	0	60	40	✅
خصم %	100	20%	0	80	✅
خصم ₪	100	20₪	50	30	✅
دفع زيادة	100	0	150	0 (بسبب Math.max)	✅

⑬ إجمالي الإيرادات — FinRevenueScreen
totalRevenue = allTxs.filter(type="in").reduce(amount)
totalDebt    = debts.reduce(amount)
totalPaid    = Math.max(0, totalRevenue - totalDebt)
collectionRate = totalRevenue > 0 ? round(totalPaid / totalRevenue × 100) : 0

⚠️ معادلة totalPaid = totalRevenue - totalDebt مفاهيمياً غير دقيقة — الإيرادات تشمل المبالغ الفعلية المدفوعة بالصندوق، والديون مُسجّلة بشكل مستقل في debts[]. الطرح ليس علاقة مباشرة رياضياً.

مثال: totalRevenue=10,000، totalDebt=3,000 → totalPaid=7,000، collectionRate=70%

القسم الثالث: حسابات الموظفين المالية
أ. العمولة getCommission
const empRevenue = sessions
  .filter(s => s.doctor === e.name && 
    (pDepts.length === 0 || pDepts.includes(s.dept)))
  .reduce((sum, s) => sum + s.paid, 0)
commission = Math.round(empRevenue × (sm.percentageValue / 100))

الشرط: salaryType ≠ "fixed" (يُرجع 0 للموظفين براتب ثابت)
المصدر: sessions[].paid × staff.percentageValue%
مثال: إيرادات الطبيب=10,000، نسبة=20% → عمولة=2,000 ✅

ب. السلف المستحقة getTotalAdvances
getTotalAdvances = (e) =>
  employeeAdvances
    .filter(a => a.empName === e.name && !a.repaid)
    .reduce((s, a) => s + a.amount, 0)

مثال: سلفتين غير مسددتين: [500, 300] → 800 ✅
ملاحظة: السلف المسددة (repaid=true) لا تُخصم

ج. صافي الراتب calcNet
calcNet = (e) => Math.max(0,
  e.salary + getCommission(e) - e.expenses - getTotalAdvances(e)
)

القيمة
راتب أساسي	1,000
+ عمولة	+ 200
− مصروفات	− 50
− سلف	− 150
صافي	= 1,000 ✅
حماية: Math.max(0, ...) → لا يصير الراتب سالباً

amounts = shares.map(s => Math.round(net × s))
